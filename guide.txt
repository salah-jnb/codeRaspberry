// initialiser environnement
source .venv/bin/activate

//cablage ecran 
RX (Jaune/) -> Pin 8 (GPIO 14 - TX) du Raspberry Pi.

TX (Bleu) -> Pin 10 (GPIO 15 - RX) du Raspberry Pi.

// checks hardware fiables (evite faux positifs)
// adapte ces valeurs a ton cablage reel
// Nextion via pins GPIO RX/TX 8/10
export NEXTION_PORT=/dev/serial0
export NEXTION_REQUIRE_GPIO_UART=1

// HC-05 de l'arduino
export HC05_MAC=00:22:12:02:35:16
// RFCOMM serie pour HC-05 (apres bind)
// sudo rfcomm bind 0 00:22:12:02:35:16
export HC05_RFCOMM_DEVICE=/dev/rfcomm0
// Regle BLE: DETECTED uniquement si /dev/rfcomm0 existe

// Audio sortie (carte ALSA exacte)
// verifie avec: aplay -l
export AUDIO_CARD_INDEX=1
// optionnel: label texte si besoin
// export AUDIO_CARD_LABEL=USB Audio Device

// lancer diagnostic
python -m app.main
